import java.util.Random;import java.util.Scanner;public class Main {    public static void main(String[] args) {        boolean campominado[][] = new boolean[5][2]; //definindo a posição da bomba        int px, py; //posições da bomba        int pxj, pyj; //posições do jogador        //comandos de criação de aleatoriedade e entrada        Random rand = new Random();        Scanner entrada = new Scanner(System.in);        //gera a posição px e py e coloca eles na posição da bomba        px = rand.nextInt(5);        py = rand.nextInt(2);        System.out.println("Bomba: "+px + " - " +py);        campominado[px][py] = true;        //inicio do game        System.out.println("Bem vindo ao seu tumulo!");        for(int i = 0;i <= campominado.length;i++){            System.out.print("posicao X: ");            pxj = entrada.nextInt();            System.out.print("posicao Y: ");            pyj = entrada.nextInt();            if(campominado[pxj][pyj])            {                System.out.println("hahaha, você morreu noob, "+"seus últimos passos foram: x="+ pxj+" y="+pyj);                break;            }            else            {                System.out.println("mais azar da próxima vez! "+"posições escolhidas: x="+ pxj+" y="+pyj);            }        }        System.out.println("!!GAME OVER!!");        entrada.close();    }}
